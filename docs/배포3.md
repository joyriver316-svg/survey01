# 서버 준비
ssh dplab-homepage 'sudo mkdir -p /var/www/auracode && sudo chown dplab:dplab /var/www/auracode'

# nginx 설정 배포 (auracode.conf 를 로컬에서 작성 후)
scp auracode.conf dplab-homepage:/tmp/
ssh dplab-homepage 'sudo cp /tmp/auracode.conf /etc/nginx/sites-available/ \
  && sudo ln -sf /etc/nginx/sites-available/auracode.conf /etc/nginx/sites-enabled/ \
  && sudo nginx -t && sudo systemctl reload nginx'

# DNS 확인 (Cafe24에서 auracode A 레코드 등록 후)
ssh dplab-homepage 'getent hosts auracode.dplab.kr'

# 배포
cd D:/projects/homepage/auracode.dplab.kr/react
npm run build
ssh dplab-homepage 'rm -rf /var/www/auracode/assets'
scp -r -q dist/. dplab-homepage:/var/www/auracode/

# HTTPS
ssh dplab-homepage 'sudo certbot --nginx -d auracode.dplab.kr \
  --non-interactive --agree-tos -m support@dplab.kr --redirect'
