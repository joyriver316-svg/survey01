# [안티그래비티 백엔드 인코딩 핵심 의사 코드]

def generate_final_skin_code(responses, scores):
    # STEP 1: 최우선 안전 필터 스크리닝 (Q9 질환 여부 체크)
    if responses['Q9'] in [1, 2, 3, 4, 5, 6]:
        skin_filter = 'SAFE_MODE_ACTIVE'
        sr = 'S'  # 총점 관계없이 무조건 강제 민감성(S) 격리 분류
    else:
        sr = 'S' if scores['S_score'] >= 30 else 'R' # 민감성 기준선 30점

    # STEP 2: 유수분 밸런스 (O/D) 판정 및 자가진단(Q10) 교차검증 알고리즘
    if scores['O_score'] >= 35 and scores['O_score'] > scores['D_score']:
        od = 'O'
    elif scores['D_score'] >= 35 and scores['D_score'] > scores['O_score']:
        od = 'D'
    else:
        # 두 행동 스코어가 모두 임계값 미만일 때 Q10 자가진단 교차 매핑 데이터로 결정
        if responses['Q10'] == 1: od = 'D'
        elif responses['Q10'] == 2: od = 'O'
        elif responses['Q10'] == 3: od = 'O'; scores['COMBO_TAG'] = True # 복합성 태그
        else: od = 'D' # 기본 디폴트 방어축 설정

    # STEP 3: 색소성(P/N) 및 주름/탄력형(W/T) 임계값 세팅
    pn = 'P' if scores['P_score'] >= 25 else 'N' # 색소성 기준선 25점
    wt = 'W' if scores['W_score'] >= 25 else 'T' # 주름/탄력 기준선 25점

    # STEP 4: 바우만 4글자 스트링 일차 조립
    bauman_code = od + sr + pn + wt  # 예시: OSPW, DRNT 등

    # STEP 5: UV 방어막 그룹 스케일 확정 (LT / MD / DP)
    uv_group = adjust_uv_by_environment(responses['Q18'], responses['Q19'], responses['latitude'])

    # STEP 6: 가중치 결합형 최종 타깃 니즈(Need) 판정 알고리즘
    # (1순위 고민 60% + 2순위 고민 30% + 현재 루틴 사용성 10%)
    need = calculate_integrated_need(responses['Q6'], responses['Q7'], responses['Q29'], responses['Q30'])

    # STEP 7: 192 분류 최종 마스터 코드 빌딩 완성
    final_combination_code = f"{bauman_code}-{uv_group}-{need}" # 예시: OSPW-MD-AA
    
    # STEP 8: 데이터 신뢰도 지수(Confidence Score) 검증 연산
    confidence_score = check_consistency_logic(responses['Q1'], responses['Q11'], responses['Q4'], responses['Q10'])
    
    return {
        "code": final_combination_code,
        "confidence": confidence_score,
        "safe_mode": skin_filter if 'skin_filter' in locals() else 'NORMAL'
    }
