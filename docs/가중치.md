🔀 PART 4. 안티그래비티 조건부 분기 라우팅 엔진 (conditional_routing.js)
안티그래비티의 [조건부 로직(Conditional Logic) 설정 탭]에 바인딩할 코드 레벨 라우팅 명세입니다. (참조 파일: '3.192분류 알고리즘.pdf')

// 안티그래비티 백엔드 타깃팅 연산 블록

function executeSkinClassification() {
    let code_bauman = "";
    let code_uv = "";
    let code_need = "";

    // 1. O/D 판정 로직 (STEP 2)
    if (O_score >= 35 && O_score > D_score) {
        code_bauman += "O";
    } else if (D_score >= 35 && D_score > O_score) {
        code_bauman += "D";
    } else {
        // 둘 다 낮을 경우 Q10 자가진단 교차 매핑 가동
        if (responses['Q10'] === 1) code_bauman += "D";
        else if (responses['Q10'] === 2) code_bauman += "O";
        else if (responses['Q10'] === 3) { code_bauman += "O"; setTag("COMBO"); }
        else code_bauman += "D";
    }

    // 2. S/R 판정 로직 (STEP 3)
    if (responses['Q9_has_value'] === true) {
        code_bauman += "S"; // 질환 의심 시 강제 진단 안전모드 락
    } else {
        code_bauman += (S_score >= 30) ? "S" : "R";
    }

    // 3. P/N 판정 로직 (STEP 4)
    code_bauman += (P_score >= 25) ? "P" : "N";

    // 4. W/T 판정 로직 (STEP 5)
    code_bauman += (W_score >= 25) ? "W" : "T";

    // 5. UV 그룹 결정 알고리즘 (STEP 6)
    let uv_raw = "";
    if (responses['Q18'] === 1 || responses['Q18'] === 2) uv_raw = "LT";
    else if (responses['Q18'] === 3 || responses['Q18'] === 4) uv_raw = "MD";
    else uv_raw = "DP";

    // 모발색(Q19) 보정 및 위도 기후 감조 자동 보정 통합 계산
    let uv_final = evaluateUVGroup(uv_raw, responses['Q19'], responses['latitude']);
    code_uv = uv_final;

    // 6. 핵심 Need 도출 다축 연산 알고리즘 (STEP 7)
    // 1순위(60%) + 2순위(30%) + 사용루틴(10%) 가중치 결합
    code_need = calculateSkinNeedScore(responses['Q6'], responses['Q7'], responses['Q29'], responses['Q30']);

    // 7. 최종 192 분류 마스터 키 코드 완성 및 출력 페이지 맵 라우팅
    let final_master_code = `${code_bauman}-${code_uv}-${code_need}`;
    
    // 신뢰도 점수(STEP 10)가 70% 미만이면 재진단 플래그 강제 레이어 렌더링
    if (calculateConfidenceScore() < 70) {
        enableFlag("RECOMMEND_RE_DIAGNOSIS");
    }

    return redirectResultPage(final_master_code);
}
